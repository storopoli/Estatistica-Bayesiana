      gaussian(\m,\s)=1/(\s*sqrt(2*pi))*exp(-((x-\m)^2)/(2*\s^2));%
      binomial(\n,\p)=\n!/(x!*(\n-x)!)*\p^x*(1-\p)^(\n-\x);%
      poisson(\l)=(\l^{x})*exp(-\l)/({x}!);%
    }
}
